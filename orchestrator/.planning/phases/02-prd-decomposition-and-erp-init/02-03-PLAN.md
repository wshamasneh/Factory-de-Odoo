<!-- Agent output schemas from CONTEXT.md -->
module-boundaries.json: { "modules": [{ "name", "description", "models": [], "base_depends": [], "estimated_complexity" }] }
oca-analysis.json: { "findings": [{ "domain", "oca_module", "odoo_module", "recommendation": "build_new"|"extend"|"skip", "reason" }] }

computation-chains.json: { "chains": [{ "name", "description", "steps": [], "cross_module": boolean }] }

<!-- Decomposition output schema -->
decomposition.json: {
  "modules": [{
    "name": string,
    "description": string,
    "models": string[],
    "base_depends": string[],
    "custom_depends": string[],
    "estimated_complexity": string,
    "build_recommendation": "build_new"|"extend"|"skip",
    "oca_module": string|null,
    "tier": string,
    "tier_index": number,
    "computation_chains": [{ "name", "description", "steps", "cross_module" }]
  }],
  "tiers": { "foundation": string[], "core": string[], ... },
  "generation_order": string[],
  "computation_chains": [{ "name", "description", "steps", "cross_module" }],
  "warnings": string[]
}

<!-- Locked ROADMAP.md format -->
```markdown
### Phase 1: uni_core
- Tier: 1 (Foundation)
- Models: uni.institution, uni.campus, uni.department, uni.program
- Depends: base, mail
- Build: NEW
- Status: not_started
```

<!-- Locked presentation format -->
```
ERP MODULE DECOMPOSITION -- {N} modules across {M} tiers

TIER 1: Foundation (generate first)
  | Module      | Models   | Build     | Depends             |
  | uni_core    | 6        | NEW       | base, mail          |

COMPUTATION CHAINS (cross-module):
  1. chain_name: step1 -> step2 -> step3

WARNINGS:
  - circular risks, ordering issues

Approve this decomposition? (yes / modify / regenerate)
```
